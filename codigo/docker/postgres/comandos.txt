copiar el backup al contenedor:
docker cp empresa3_pg.sql nombre_contenedor:/tmp/empresa3_pg.sql

# Crear el contenedor:
docker-compose up -d


# comcprobar si esta arrancado:
docker ps


# Conectar con el contenedor: 
docker exec -it postgres_dev bash


# Como el backup es binario, hay que ejecutarlo con pg_restore:
# OJO meter el role, porque el backup se hizo con el user: postgre
pg_restore -U nombre_user -d empresa3 --role=nombre_user /tmp/empresa3_pg.sql


# Conectar con psql:
psql -U nombre_user -d nombre_bd

\l  listar BDs
\dt listar tablas
select * from tbempleados;
